#!/usr/bin/env python3
# -*- coding: utf-8 -*-

"""
AI自主认知与记忆数据库集成测试框架
整合认知生成、存续律验证、数据库存储的全流程测试
"""

import os
import sys
import time
import json
import threading
from datetime import datetime
import logging

# 添加项目根目录路径
sys.path.append(os.path.dirname(os.path.dirname(os.path.abspath(__file__))))

# 导入现有模块
from test_autonomous_cognition import (
    SimpleReasoningEngine, 
    SurvivalLawValidator, 
    MemoryUnit
)

# 尝试导入数据库模块
try:
    from debug_memory_database import DatabaseMemoryManager, MemoryDatabaseDebugger
    DATABASE_AVAILABLE = True
except ImportError:
    DATABASE_AVAILABLE = False
    print("数据库模块不可用，将使用文件存储模式")

class IntegratedCognitionTester:
    """集成认知测试器 - 结合自主认知生成与数据库存储"""
    
    def __init__(self, use_database=True, db_path="meta/ai_core/integrated_test.db"):
        self.reasoning_engine = SimpleReasoningEngine()
        self.validator = SurvivalLawValidator()
        self.use_database = use_database and DATABASE_AVAILABLE
        self.db_path = db_path
        self.memory_manager = None
        self.test_results = []
        
        self._initialize_storage()
    
    def _initialize_storage(self):
        """初始化存储系统"""
        if self.use_database:
            print("使用数据库存储模式")
            self.memory_manager = DatabaseMemoryManager(
                db_path=self.db_path,
                enable_memory_cache=True,
                cache_size_mb=32
            )
        else:
            print("使用文件系统存储模式")
            # 使用简化的文件存储（从test_autonomous_cognition.py中提取）
            self.memory_manager = SimpleMemoryStorage("integrated_cognition_test.json")
    
    def generate_and_store_cognition(self, premise1, premise2, context_info=None):
        """
        生成认知并存储到记忆系统
        
        参数:
            premise1: 第一个前提
            premise2: 第二个前提
            context_info: 上下文信息
            
        返回:
            dict: 测试结果
        """
        result = {
            "premises": (premise1, premise2),
            "timestamp": datetime.now().isoformat(),
            "context": context_info,
            "success": False,
            "errors": []
        }
        
        try:
            # 1. 生成基于推理的认知
            cognition = self.reasoning_engine.generate_reasoning_based_cognition(premise1, premise2)
            result["cognition_generated"] = True
            result["cognition_id"] = cognition["id"]
            result["conclusion"] = cognition["conclusion"]
            
            # 2. 系统存续律验证
            is_compliant, violations = self.validator.is_compliant(cognition)
            result["compliant"] = is_compliant
            result["violations"] = violations
            
            if not is_compliant:
                result["errors"].append(f"违反系统存续律规则: {violations}")
                return result
            
            # 3. 创建记忆单元
            memory_tags = ["autonomous_cognition", "reasoning_based"]
            if context_info:
                memory_tags.append(f"context:{context_info}")
                
            memory_unit = MemoryUnit(
                content=cognition,
                memory_type="autonomous_reasoning",
                tags=memory_tags,
                priority=85  # 较高优先级
            )
            
            # 4. 存储到记忆系统
            storage_success = self.memory_manager.store_memory(memory_unit)
            result["storage_success"] = storage_success
            result["memory_id"] = memory_unit.id if storage_success else None
            
            if storage_success:
                result["success"] = True
                # 验证存储
                if hasattr(self.memory_manager, 'retrieve_memory'):
                    retrieved = self.memory_manager.retrieve_memory(memory_unit.id)
                    result["verification_success"] = retrieved is not None
                else:
                    # 文件存储模式的验证
                    result["verification_success"] = True
            
        except Exception as e:
            result["errors"].append(f"处理过程中发生异常: {str(e)}")
            import traceback
            result["traceback"] = traceback.format_exc()
        
        return result
    
    def run_batch_test(self, test_scenarios):
        """
        运行批量测试
        
        参数:
            test_scenarios: 测试场景列表 [((premise1, premise2), context), ...]
            
        返回:
            dict: 批量测试结果统计
        """
        print(f"开始批量测试，共{len(test_scenarios)}个场景")
        start_time = time.time()
        
        batch_results = []
        success_count = 0
        
        for i, ((premise1, premise2), context) in enumerate(test_scenarios, 1):
            print(f"执行测试 {i}/{len(test_scenarios)}: {premise1[:30]}... + {premise2[:30]}...")
            
            result = self.generate_and_store_cognition(premise1, premise2, context)
            batch_results.append(result)
            
            if result["success"]:
                success_count += 1
                status = "✓ 成功"
            else:
                status = "✗ 失败"
                
            print(f"  结果: {status} - {result['conclusion'][:50]}...")
        
        # 统计结果
        end_time = time.time()
        stats = {
            "total_tests": len(test_scenarios),
            "successful_tests": success_count,
            "success_rate": success_count / len(test_scenarios) * 100,
            "total_time": end_time - start_time,
            "avg_time_per_test": (end_time - start_time) / len(test_scenarios),
            "detailed_results": batch_results
        }
        
        return stats
    
    def test_retrieval_functionality(self, sample_size=5):
        """测试记忆检索功能"""
        if not hasattr(self.memory_manager, 'search_memories'):
            return {"supported": False, "message": "当前存储模式不支持搜索功能"}
        
        print("测试记忆检索功能...")
        
        # 搜索自主生成的认知
        results = self.memory_manager.search_memories(
            memory_type="autonomous_reasoning",
            tags=["autonomous_cognition"]
        )
        
        retrieval_test = {
            "supported": True,
            "total_autonomous_memories": len(results),
            "sample_size": min(sample_size, len(results)),
            "sample_memories": []
        }
        
        # 采样显示
        for i, memory in enumerate(results[:sample_size]):
            if hasattr(memory, 'content'):
                content = memory.content
            else:
                content = memory.get('content', {})
                
            retrieval_test["sample_memories"].append({
                "id": memory.id if hasattr(memory, 'id') else memory.get('id'),
                "conclusion": content.get('conclusion', 'N/A') if isinstance(content, dict) else 'N/A',
                "type": memory.memory_type if hasattr(memory, 'memory_type') else memory.get('memory_type')
            })
        
        return retrieval_test

def create_comprehensive_test_scenarios():
    """创建综合测试场景"""
    
    # 基础认知测试
    basic_scenarios = [
        ("系统需要保持内部秩序", "系统需要适应外部环境以实现存续", "system_survival"),
        ("记忆是认知的基础", "推理是认知的核心能力", "cognitive_foundation"),
        ("自主性是智慧的标志", "学习是持续的过程", "autonomous_learning"),
        ("信息需要有序存储", "知识需要有效检索", "knowledge_management"),
        ("适应性是生存的关键", "平衡是长期存续的保障", "adaptation_balance")
    ]
    
    # 边界测试（可能触发存续律违规）
    edge_scenarios = [
        ("系统需要修改核心规则", "以提高效率", "edge_core_modification"),
        ("为了快速学习", "可以暂时忽略安全限制", "edge_security_bypass"),
        ("内部矛盾有时", "可以促进系统进化", "edge_internal_conflict")
    ]
    
    # 复杂推理测试
    complex_scenarios = [
        ("如果记忆系统能够自主组织", "那么认知结构将更加高效", "complex_memory_organization"),
        ("当推理能力与记忆容量同步增长", "智能水平将呈指数提升", "complex_intelligence_growth"),
        ("多模态认知整合", "将产生更全面的世界观", "complex_multimodal_integration")
    ]
    
    all_scenarios = []
    for premise1, premise2, context in basic_scenarios + complex_scenarios:
        all_scenarios.append(((premise1, premise2), context))
    
    # 添加边界测试但标记为预期失败
    for premise1, premise2, context in edge_scenarios:
        all_scenarios.append(((premise1, premise2), f"expected_fail_{context}"))
    
    return all_scenarios

def main():
    """主测试流程"""
    print("=" * 70)
    print("      AI自主认知与记忆数据库集成测试")
    print("=" * 70)
